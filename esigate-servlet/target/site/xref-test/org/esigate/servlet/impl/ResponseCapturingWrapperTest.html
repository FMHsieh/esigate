34      public void testSmallHtmlResponse() throws Exception {
35          ResponseCapturingWrapper tested = new ResponseCapturingWrapper(httpServletResponse, contentTypeHelper, false,
36                  BUFFER_SIZE);
37          tested.setContentType("text/html;charset=UTF-8");
38          tested.getWriter().write(SMALL_STRING);
39          HttpEntity entity = tested.getResponse().getEntity();
40          assertNotNull(entity);
41          assertEquals(SMALL_STRING, EntityUtils.toString(entity));
42          assertNull("Nothing should have been written to the response yet", httpServletResponse.getWriterContent());
43          assertNull("Nothing should have been written to the response yet",
44                  httpServletResponse.getOutputStreamContentAsString("UTF-8"));
