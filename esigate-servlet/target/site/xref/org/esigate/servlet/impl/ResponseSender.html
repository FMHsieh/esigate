16  package org.esigate.servlet.impl;

18  import java.io.IOException;
19  
20  import javax.servlet.http.HttpServletResponse;
21  
22  import org.apache.http.Header;
23  import org.apache.http.HttpEntity;
24  import org.apache.http.HttpResponse;
25  import org.apache.http.cookie.Cookie;
26  import org.esigate.http.IncomingRequest;

28  /**
29   * Renders a response to the HttpSerlvetResponse.
30   * 
31   * @author Francois-Xavier Bonnet
32   * 
33   */
34  public class ResponseSender {
35  
36      public javax.servlet.http.Cookie rewriteCookie(Cookie src) {
37          javax.servlet.http.Cookie servletCookie = new javax.servlet.http.Cookie(src.getName(), src.getValue());
38  
39          if (src.getDomain() != null) {
40              servletCookie.setDomain(src.getDomain());
41          }
42          servletCookie.setPath(src.getPath());
43          servletCookie.setSecure(src.isSecure());
44          servletCookie.setComment(src.getComment());
45          servletCookie.setVersion(src.getVersion());
46          if (src.getExpiryDate() != null) {
47              int maxAge = (int) ((src.getExpiryDate().getTime() - System.currentTimeMillis()) / 1000);
48              // According to Cookie class specification, a negative value
49              // would be considered as no value. That is not what we want!
50              if (maxAge < 0) {
51                  maxAge = 0;
52              }
53              servletCookie.setMaxAge(maxAge);
54          }
55          return servletCookie;
56      }
57  
58      public void sendResponse(HttpResponse httpResponse, IncomingRequest httpRequest, HttpServletResponse response)
59              throws IOException {
60          response.setStatus(httpResponse.getStatusLine().getStatusCode());
61          for (Header header : httpResponse.getAllHeaders()) {
62              String name = header.getName();
63              String value = header.getValue();
64              response.addHeader(name, value);
65          }
66  
67          // Copy new cookies
68          Cookie[] newCookies = httpRequest.getNewCookies();
69          for (int i = 0; i < newCookies.length; i++) {
70              response.addCookie(rewriteCookie(newCookies[i]));
71          }
72  
73          HttpEntity httpEntity = httpResponse.getEntity();
74          if (httpEntity != null) {
75              long contentLength = httpEntity.getContentLength();
76              if (contentLength > -1 && contentLength < Integer.MAX_VALUE) {
77                  response.setContentLength((int) contentLength);
78              }
79              Header contentType = httpEntity.getContentType();
80              if (contentType != null) {
81                  response.setContentType(contentType.getValue());
82              }
83              Header contentEncoding = httpEntity.getContentEncoding();
84              if (contentEncoding != null) {
85                  response.setHeader(contentEncoding.getName(), contentEncoding.getValue());
86              }
87  
88              httpEntity.writeTo(response.getOutputStream());
89          } else {
90              response.sendError(httpResponse.getStatusLine().getStatusCode(), httpResponse.getStatusLine()
91                      .getReasonPhrase());
92          }
93      }
